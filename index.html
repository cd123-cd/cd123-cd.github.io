class HtmlOutputer(object):
def init(self):
self.datas=[]

def collect_data(self,data):
    if data is None:
        return
    self.datas.append(data)
def output_html(self):
    fout=open('output.html','w',encoding='utf-8')
    fout.write("<html>")
    fout.write("<meta charset='utf-8'>")
    fout.write("<body>")
    fout.write("<table>")
    for data in self.datas:
        fout.write("<tr>")
        fout.write("<td>%s</td>" % data['url'])
        fout.write("<td>%s</td>" % data['title'])
        fout.write("<td>%s</td>" % data['summary'])
        fout.write("</tr>")
    fout.write("</table>")
    fout.write("</body>")
    fout.write("</html>")
    fout.close()

爬虫（1）

简单爬虫架构
爬虫调度端->URL管理器->网页下载器->网页解析器->价值数据
网页解析器再返回URL管理器
简单爬虫架构-运行流程
调度器询问管理器有没有待爬URL，URL返回是和否，是->URL管理器会从调度器中取出一个待爬取URL，URL管理器会从中返回URL给调度器，调度器将URL传给下载器，下载器会将下载URL内容，然后返回给调度器，调度器再将内容传给解析器，解析器解析后会将新的URL和价值数据返回给调度器，然后调度器会将收集到的价值数据传送给应用，另一方面，会将新增的待爬取URL补充给URL管理器输出价值数据
